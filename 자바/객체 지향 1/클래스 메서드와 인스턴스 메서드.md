## 클래스 메서드와 인스턴스 메서드


1. 인스턴스 메서드 (비정적 메서드)  해당 변수에 담는 과정을 인스턴스라고 한다.
2. 클래스 메서드 (정적 메서드)

<br/>

간단히 말해 인스턴스 메서드는 Static을 붙이지 않고 선언한 메서드이고,

클래스 메서드는 static을 붙여 선언한 메서드 입니다.

인스턴스 변수 == 인스턴스 필드

클래스 변수 == 스테틱 필드
<br/><br/>

    인스턴스 메서드
    static을 붙여 있지 않은 것들은 인스턴스 메서드 이다. 
    → 클래스 맴버와 클래스 변수에 접근(호출) 가능

    클래스 메소드
    인스턴스 맴버와 인스턴스 변수에 접근 할 수 없다.

    
    즉, 이유가 뭐냐면
    밑에 그림 처럼 클래스가 위에 있고 밑에 인스턴스가 있다
    
    인스턴스를 만들려면 당연히 위에 있는 클래스가 필요하기 때문에    
    인스턴스는 당연히 클래스를 알고 있다(접근 가능하단 소리)
    
    반대로 클래스에서 인스턴스에 접근 하려 하면 인스턴스가 
    아직 만들어져 있는 상태인지? 아닌지?를 모르기 때문에 
    접근 할수가 없는것이다.

```java
package ddd;

public class Call {

	static int static_variable = 1; // 클래스 변수
	int intstance_variable = 2; // 인스턴스 변수

	static void static_static() { // 클래스 메서드가 클래스 변수에 접근 할때
		System.out.println(static_variable); // 클래스 메서드에서 println안에를 있는것은
                                         // 클래스 변수이기 때문에 가능한것이다.
	}

	static void static_instance() { // 클래스 메서드가 인스턴스 변수에 접근 할때
	                    	// 클래스 메소드에서는 인스턴스 변수에 접근 할 수 없다.
		// System.out.println(instance_variable); // 주석 처리 되어 있는것을 풀어보면 
       // 에러가 발생한다. 이유는 println안에 있는것은 인스턴스 변수이기 때문에 
	}

	void instance_static() {
		// 인스턴스 메소드에서는 클래스 변수에 접근 할 수 있다.
		System.out.println(static_variable);
	}

	void instance_instance() {
		System.out.println(intstance_variable);
	}

	public static void main(String[] args) {

		Call c = new Call();
		// 인스턴스를 이용해서 정적 메소드에 접근 -> 성공
		// 인스턴스 메소드가 정적 변수에 접근 -> 성공
		
		c.static_static();
		// 인스턴스를 이용해서 정적 메소드에 접근 -> 성공
		// 정적 메소드가 인스턴스 변수에 접근 -> 실패
		
		c.static_instance();
		// 인스턴스를 이용해서 인스턴스 메소드에 접근 -> 성공
		// 인스턴스 메소드가 클래스 변수에 접근 -> 성공
		
		c.instance_static();
		// 인스턴스를 이용해서 인스턴스 메소드에 접근 - > 성공
		// 인스턴스 메소드가 인스턴스 변수에 접근 -> 성공
		
		c.instance_instance();
		// 클래스를 이용해서 클래스 메소드에 접근 -> 성공
		// 클래스 메소드가 클래스 변수에 접근 -> 성공
		
		Call.static_static();
		// 클래스를 이용해서 클래스 메소드에 접근 -> 성공
		// 클래스 메소드가 인스턴스 변수에 접근 -> 실패
		
		Call.static_instance();
		// 클래스를 이용해서 인스턴스 메소드에 접근 -> 실패 (위쪽에 '에러 발생' 이유 잘보기)
		// Call.instance_static(); //클래스에서 인스턴스 메서드로 갈수가 없다
		// 클래스를 이용해서 인스턴스 메소드에 접근 -> 실패
		// Call.instance instance(); //클래스에서 인스턴스 메서드로 갈수가 없다
		
		
	}
}
```

## 자바의 시작은 메인 부터 시작하여 내려가는 것이다.
```java
package test;

public class titi {

	private static int counter = 0; // 아이디를 몇 개 부여했는지 저장 (클래스) -- (두번째)
	private int id; // 아이디 (인스턴스) -- (두번째) 

	// 생성자
	public titi() { // 순서 체크하기 -- (첫번째)
		id = ++counter;
	}

	// 아이디를 반환하는 인스턴스 메서드
	public int getId() { // -- (세번째)
		return id;
	}

	// counter를 반환하는 클래스 메서드
	public static int getCounter() {
		return counter;
	}

	public static void main(String[] args) { // 자바에서의 시작은 여기서 부터이다 

		titi a = new titi(); // 아이디 1 (인스턴스 만들기) 즉, titi 클래스 안에 있는 메서드를 
												 // 사용하려고 인스턴스를 만드는 것이다

		titi b = new titi(); // 아이디 2 (인스턴스 만들기)

		System.out.println("a의 아이디: " + a.getId()); 
			 // a라는 클래스안에 getId()메서드를 사용한다는 뜻 
			 // (a는 titi 클래스안에 있는 것을 가져와서 사용해야하기 때문에 a. 이라고 해주는 것이다)
		
		System.out.println("b의 아이디: " + b.getId());

		System.out.println("부여한 아이디의 개수:" + titi.getCounter());

	}

}

출력 값.
a의 아이디: 1
b의 아이디: 2
부여한 아이디의 개수: 2

```
